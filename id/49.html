    <title>自定义TABLE生成</title>

        .baseinput{

</head>
<body>
    <input type="text" placeholder="请输入TABLE公式" id="tablemath">
    <input type="button" value="生成" onclick="CreateTable()"><br/><br/>
    <b>如何自定义一个表格？</b><br/>
    C=5*6&nbsp;创建一个5行6列表格
    <br/><br/>
    
    <b>如何合并自定义表格列？</b><br/>
    L=1*1~3&nbsp;合并第1行的第1列至第3列<br/>
    <span style="color:red; font-size: 14px;">注意：如果当前行下的列要多次合并，那么之前合并过的列==1</span><br/>
    L=1*2~4&nbsp;合并第1行的第2列至第4列<br/><br/>
    
    <b>如何合并自定义表格行？</b><br/>
    R=1*2~3&nbsp;合并第1列的第2行至第3行<br/>
    <span style="color:red; font-size: 14px;">注意：如果当前列下的行要多次合并，那么之前合并过的行数算做原有的行数</span><br/>
    R=1*4~5&nbsp;合并第1列的第4行至第5行<br/><br/>
    
    <b>如何初始化自定义表格行？</b><br/>
    重新生成表格<br/>
    DR=1 初始化第一行<br/>
    DR=1~5 初始化第一行至第五行
    <br/><br/>
    
    如有需要请自行Ctrl+U
    
    <div id="content"></div>

    JS：

    <script>
/**
 * 自定义TABLE
 *
 * @authors 刘 贝 (993008916@qq.com)
 * @date    2017-04-11 12:44:48
 * @version v1.0
 */
// 初始化table
var tableStartStr = '<table border="1" width="100%" id="tab_Forms">';
var tableEndStr   = '</table>';
var tableTdStr    = '<td><input type="text" class="baseinput" onpropertychange="vals(this)" oninput="vals(this)"></td>';
var tableRows,tableCols,math;
// CreateTable();
function CreateTable()
{
// 获取公式字符串
var tableMath = $.trim($('#tablemath').val());
//console.log(tableMath)
    if(tableMath == '')
    {
        alert('TABLE公式不能为空');
        return false;
    }
    // 获取公式前缀
    var tablePrefix = tableMath.split('=')[0];
    var tableType = tableMath.split('=')[1];
    // console.log(tablePrefix);
    // console.log(tableType);
    // C=创建表格
    if(tablePrefix === 'C')
    {
        // 获取表格行列
        var rowCount = parseInt(tableType.split('*')[0]);
        var colCount = parseInt(tableType.split('*')[1]);
        tableRows = rowCount;
        tableCols = colCount;
        console.log('创建一个'+rowCount+'行'+colCount+'列的表格');
        // 构造表格头 创建列值
        var tempStr = tableStartStr;
        // 循环tr
        for(var i=0; i<rowCount; i++)
        {
            tempStr += '<tr>';
            // 循环td
            for(var j=0; j<colCount; j++)
            {
                tempStr += tableTdStr;
            }
            tempStr += '</tr>';
        }
        // 构造表格尾
        tempStr += tableEndStr;
        // 放入指定节点
        $('#content').empty().append(tempStr);
        return false;
    }
    // L=合并列
    if(tablePrefix === 'L')
    {
    var globalnum = parseInt(tableType.split('*')[0]);
        var tdStart   = parseInt(tableType.split('*')[1].split('~')[0]);
        var tdEnd     = parseInt(tableType.split('*')[1].split('~')[1]);
        //console.log(globalnum+'---'+tdStart+'---'+tdEnd);
        console.log('合并'+ '第' + globalnum + '行的'+'第'+ tdStart+'列到第'+tdEnd+'列');
        // 查询当前行下的列是否有过合并
        var isColSpanErr = 0;
      $('#tab_Forms').find('tr').eq(globalnum).find('td').each(function(tdindex){
      var isColSpan = $(this).attr('colspan');
      if(isColSpan == undefined)
      {
      isColSpanErr++;
      }
        });
        //console.log(isColSpanErr +'---'+ tableCols)
        if (isColSpanErr === tableCols)
        {
        var isColSpan = false;
        }
        else
        {
        var isColSpan = true;
        }
        console.log(isColSpan);
        $('#tab_Forms').find('tr').eq(globalnum-1).find('td').each(function(tdindex){
    //console.log($(this));
    // 起始合并位置
    if (tdindex == (tdStart-1))
    {
    var diff = tdEnd - tdStart + 1;
    $(this).attr('colspan',diff);
    }
    // 最后/中间位置删除
    else if(tdindex > (tdStart-1) && tdindex <= (tdEnd-1))
    {
    $(this).remove();
    }
    });
    }
    // R=合并行
    if(tablePrefix === 'R')
    {
    var globalnum = tableType.split('*')[0];
        var tdStart = tableType.split('*')[1].split('~')[0];
        var tdEnd   = tableType.split('*')[1].split('~')[1];
        //console.log(globalnum+'---'+tdStart+'---'+tdEnd);
        console.log('合并'+ '第' + globalnum + '列的'+'第'+ tdStart+'行到第'+tdEnd+'行');
 // 遍历tr
 var isRowSpanErr = 0;
        $('#tab_Forms').find('tr').each(function(trindex){
        // 查询当前列下的行是否有过合并
        var isRowSpan = $(this).find('td').eq(globalnum-1).attr('rowspan');
        if(isRowSpan == undefined)
        {
        isRowSpanErr++;
        }
        //console.log(isRowSpan);
        });
        if (isRowSpanErr === tableRows)
        {
        var isRowSpan = false;
        }
        else
        {
        var isRowSpan = true;
        }
        $('#tab_Forms').find('tr').each(function(trindex){
    // 找到合并起始行在指定列追加rowspan
    if(trindex == (tdStart-1))
    {
    var diff = tdEnd - tdStart + 1;
    $(this).find('td').eq(globalnum-1).attr('rowspan',diff);
    }
    else if(trindex > (tdStart-1) && trindex <= (tdEnd-1))
    {
    $(this).find('td').eq(globalnum-1).remove();
    }
    });
    }
    //DR=初始化多行
   if(tablePrefix === 'DR')
   {
   var tdStart = tableType.split('~')[0];
        var tdEnd   = tableType.split('~')[1];
    // DL=初始化1行
    if(tdEnd == undefined)
    {
    console.log('初始化第'+tableType+'行');
    $('#tab_Forms').find('tr').eq(tableType-1).empty();
    for(var i=0; i<tableCols; i++)
    {
    $('#tab_Forms').find('tr').eq(tableType-1).append(tableTdStr);
    }
    }
    else
    {
        console.log('初始化'+'第'+ tdStart+'行到第'+tdEnd+'行');
        $('#tab_Forms').find('tr').each(function(trindex){
    if(trindex >= (tdStart-1) && trindex <= (tdEnd-1))
    {
    $(this).empty();
    for(var i=0; i<tableCols; i++)
    {
    $(this).append(tableTdStr);
    }
    }
    });
    }
   }
}
function vals(_this)
{
$(_this).attr('value',$(_this).val());
}
</script>
